def planeStrainNeoHookean( props, F ): # Props - poisson, E
I2 = F*0.
I2[0,0] = I2[1,1] = 1.
v = props['poisson']
E = props['modulus']
l = E * v / ((1.+v)*(1.-2.*v))
m = 0.5 * E / (1.+v)
Ja = F[0,0]*F[1,1] - F[1,0]*F[0,1]
S = I2*l*np.log(Ja)/Ja + m/Ja * (np.dot(F, F.T) - I2)
return (S,Ja)
